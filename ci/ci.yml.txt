name: ci

on:
  push:
    branches: [main]
  pull_request:

jobs:
  test-and-run-notebooks:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
        with:
          python-version: "3.12"
      - name: Install package + notebook/dev extras
        run: |
          python -m pip install --upgrade pip
          python -m pip install -e ".[notebooks,dev]"
      - name: Run tests
        run: pytest -q
      - name: Execute every chapter notebook
        run: |
          for nb in notebooks/*.py; do
            echo "::group::$nb"
            jupytext --to notebook --execute "$nb" --output /tmp/_out.ipynb
            echo "::endgroup::"
          done
